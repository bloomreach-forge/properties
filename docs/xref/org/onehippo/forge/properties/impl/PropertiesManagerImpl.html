74      @Override
75      public PropertiesBean getPropertiesBean(final String path, final HippoBean baseBean, final Locale locale) {
76          if (path == null) {
77              // get document by default name
78              return getPropertiesBean(baseBean, locale);
79          } else {
80              // get document by given path
81              final HippoBean location = getDefaultLocation(baseBean);
82              return getPropertiesBean(location, path, locale);
83          }
84      }

86      /** {@inheritDoc} */
87      @Override
88      public List<PropertiesBean> getPropertiesBeans(final List<String> paths, final HippoBean baseBean, final Locale locale) {
89  
90          List<PropertiesBean> propertiesBeans = new ArrayList<PropertiesBean>(paths.size());
91  
92          if (paths.size() == 0) {
93              // get document by default name
94              final PropertiesBean propertiesBean = getPropertiesBean(baseBean);
95              if (propertiesBean != null) {
96                  propertiesBeans.add(propertiesBean);
97              }
98          } else {
99              // get multiple documents by given paths
100             final HippoBean location = getDefaultLocation(baseBean);
101             for (final String path : paths) {
102 
103                 final PropertiesBean propertiesBean = getPropertiesBean(location, path, locale);
104                 if (propertiesBean != null) {
105                     propertiesBeans.add(propertiesBean);
106                 }
107             }
108         }
109 
110         return propertiesBeans;
111     }

130     /**
131      * Get the base location where the properties documents are stored.
132      *
133      * @param baseBean normally the site content base bean
134      * @return default location bean
135      */
136     protected HippoBean getDefaultLocation(HippoBean baseBean) {
137 
138         if (baseBean == null) {
139             throw new IllegalArgumentException("argument 'baseBean' is null");
140         }
141         if (this.defaultDocumentLocation == null) {
142             throw new IllegalStateException("defaultDocumentLocation is null: " + this.getClass().getSimpleName() + " not correctly configured");
143         }
144 
145         HippoBean defaultLocation;
146         if (this.defaultDocumentLocation.startsWith("/")) {
147 
148             // support for location outside of site content root
149             Object absoluteLocation;
150             try {
151                 absoluteLocation = baseBean.getObjectConverter().getObject(baseBean.getNode().getSession(), this.defaultDocumentLocation);
152 
153             } catch (Exception e) {
154                 throw new IllegalStateException("Can't get location by path " + this.defaultDocumentLocation, e);
155             }
156 
157             if (!(absoluteLocation instanceof HippoBean)) {
158                 throw new IllegalStateException("Default location object by path " + this.defaultDocumentLocation
159                         + " is not a HippoBean but " + ((absoluteLocation == null) ? "null" : absoluteLocation.getClass().getName()));
160             }
161 
162             defaultLocation = (HippoBean) absoluteLocation;
163         } else {
164             defaultLocation = baseBean.getBean(this.defaultDocumentLocation);
165         }
166 
167         if (defaultLocation == null) {
168             throw new IllegalStateException("Default location '" + this.defaultDocumentLocation +
169                     "' (possibly relative to " + baseBean.getPath() + ") is not a folder in the repository");
170         }
171 
172         return defaultLocation;
173     }

175     /**
176      * Get a serializable PropertiesBean by location, path and locale.
177      *
178      * @param location default location where to find properties beans
179      * @param path     path relative to the location for a particular bean
180      * @param locale   locale by which to find linked properties documents
181      * @return Serializable cacheable properties bean, based on a properties document
182      */
183     protected PropertiesBean getPropertiesBean(final HippoBean location, final String path, final Locale locale) {
184         if (location == null) {
185             throw new IllegalArgumentException("Location bean is null, path=" + path);
186         }
187         if (path == null) {
188             throw new IllegalArgumentException("Path is null, location bean is " + location.getPath());
189         }
190 
191         final Properties doc = getTranslatedProperties(location, path, locale);
192         if (doc != null) {
193 
194             return new PropertiesBean(doc);
195         }
196         return null;
197     }
198 
199     protected Properties getTranslatedProperties(final HippoBean location, final String path, final Locale locale) {
200 
201         Properties propertiesDoc = location.getBean(path, Properties.class);
202 
203         // check availability of translations in a preferred locale
204         if ((propertiesDoc != null) && (locale != null)) {
205             HippoAvailableTranslationsBean<Properties> translationBean = propertiesDoc.getAvailableTranslations(Properties.class);
206             if (translationBean != null) {
207                 // first try full locale, then just the language
208                 if (translationBean.hasTranslation(locale.toString())) {
209                     propertiesDoc = translationBean.getTranslation(locale.toString());
210                 }
211                 else if (translationBean.hasTranslation(locale.getLanguage())) {
212                     propertiesDoc = translationBean.getTranslation(locale.getLanguage());
213                 }
214             }
215         }
216 
217         return propertiesDoc;
218     }
